names, means, stds = [], [], []

for name, model in models.items():
    
    pipe = Pipeline([
        ("imputer", SimpleImputer(strategy="mean")),
        ("scaler", StandardScaler()),
        ("model", model)
    ])
    
    scores = cross_val_score(
